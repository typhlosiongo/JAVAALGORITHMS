public class Foothill
{
   public static void main(String[] args)
   {
      // arrays to be tested
      int testArray = 2000;
      quickSort(testArray);
      testArray = 200000;
      quickSort(testArray);
      testArray = 500000;
      quickSort(testArray);
      testArray = 1000000;
      quickSort(testArray);
      testArray = 1500000;
      quickSort(testArray);
      testArray = 2000000;
      quickSort(testArray);
   }

   // uses fhsort to order the arrays as well as time the process
   public static void quickSort(int arraySize)
   {
      int i, recursionLimit, rand;
      double start, finish;
      Integer[] numbers = new Integer[arraySize];
      Integer[] ordered = new Integer[arraySize];

      for (i = 0; i < arraySize; i++)
      {
         rand = (int) (Math.random() * arraySize);
         numbers[i] = rand;
      }

      System.out.println("Array Size = " + arraySize);
      System.out.println("Recursive Limit | Time (Seconds)");
      System.out.println("------------");
      
      //recursion for loop
      for (recursionLimit = 2; recursionLimit < 302; recursionLimit += 2)
      {
         for (i = 0; i < arraySize; i++)
         {
            ordered[i] = numbers[i];
         }
         //timing mechanism
         FHsort.setRecursionLimit(recursionLimit);
         start = System.nanoTime();

         FHsort.quickSort(ordered);
         finish = System.nanoTime();
         
         System.out.print(recursionLimit + "   |   ");
         System.out.printf("%.4f %n", (finish - start) * Math.pow(10, -9));
      }
      System.out.println("------------");
   }
}
